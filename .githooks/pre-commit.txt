#!/usr/bin/env sh
set -e

echo ""
echo "[pre-commit] Android build check started..."
echo ""

# logs 폴더나 APK가 실수로 커밋되는지 차단
if git diff --cached --name-only | grep -E '(^logs/|\.log$|\.apk$|\.aab$|local\.properties$)' > /dev/null; then
  echo "[pre-commit] ERROR: logs, APK/AAB, local.properties must not be committed."
  echo ""
  git diff --cached --name-only | grep -E '(^logs/|\.log$|\.apk$|\.aab$|local\.properties$)' || true
  echo ""
  exit 1
fi

# Android 빌드 검증
./gradlew.bat :app:assembleDebug

echo ""
echo "[pre-commit] Build passed. Commit allowed."
echo ""
